Core Architecture Principles

Component-Based Architecture

UI Components

Located in src/components/ui/ - Built on Radix UI primitives

  • Reusable, accessible components
  • Consistent styling with Tailwind CSS
  • TypeScript interfaces for all props

Feature Components

Direct children of src/components/

  • Business logic components
  • Firebase data integration
  • Modal and form components

Analytics Components

Located in src/components/analytics/

  • Advanced chart visualizations

Design Patterns

Data Flow Pattern

// Expense Management Flow - Complete CRUD pattern used throughout app
interface Expense {
  id: string;          // Firestore document ID
  amount: number;      // Expense amount in user's currency
  category: string;    // Category name (from DEFAULT_CATEGORIES or custom)
  description: string; // User-provided description
  date: string;        // Date in YYYY-MM-DD format
  createdAt: string;   // ISO timestamp for creation time
  receiptUrl?: string; // Optional Firebase Storage URL for receipt image
  peopleIds?: string[]; // Array of person IDs this expense was shared with
}

// Firebase Operations Pattern - Standard error handling for all operations
const addExpense = async (expense: Omit<Expense, 'id'>) => {
  // 1. Validate user authentication (CRITICAL - all operations must check this)
  if (!user) {
    toast.error("Please sign in to continue");
    return;
  }
  
  // 2. Add to Firestore with comprehensive error handling
  try {
    await addExpenseToFirestore(user.uid, expense);
    toast.success("Expense added successfully");
    // Local state automatically updates via onSnapshot subscription
  } catch (error) {
    console.error('Add expense failed:', error);
    toast.error("Failed to add expense");
    // Error is logged but app continues functioning
  }
};

ApexCharts Architecture (Critical)

Important: ApexCharts requires careful lifecycle management in React, especially with tab switching.

Chart Lifecycle Management

// CRITICAL: Tab switching pattern for charts
// ApexCharts in React requires careful DOM lifecycle management
const [activeTab, setActiveTab] = useState('overview');
const [chartKey, setChartKey] = useState(0); // Forces React re-render

// Chart instance refs for proper cleanup to prevent memory leaks
const areaChartInstance = useRef<ApexCharts | null>(null);
const columnChartInstance = useRef<ApexCharts | null>(null);

// Force re-render for chart visibility - ESSENTIAL for ApexCharts
const handleTabChange = (value: string) => {
  setActiveTab(value);
  if (value === 'overview') {
    // This increment forces useEffect to re-run and recreate charts
    setChartKey(prev => prev + 1); // REQUIRED - charts disappear without this
  }
};

// Chart cleanup and initialization pattern
useEffect(() => {
  // 1. Clean up existing chart instances to prevent memory leaks
  if (areaChartInstance.current) {
    areaChartInstance.current.destroy();
    areaChartInstance.current = null;
  }
  
  // 2. Re-initialize when conditions are met
  if (activeTab === 'overview' && areaChartRef.current && data.length > 0) {
    const chart = new ApexCharts(areaChartRef.current, chartConfig);
    chart.render();
    areaChartInstance.current = chart; // Store reference for cleanup
  }
}, [activeTab, chartKey, expenses, budgets]); // Include all dependencies

// Chart containers MUST have unique keys for re-rendering
// This key change forces DOM element recreation when chartKey changes
<div 
  key={`area-chart-${chartKey}`}  // Unique key prevents stale DOM references
  ref={areaChartRef} 
  className="w-full h-64"
/>

Security Rules Pattern

// Firestore Security Rules - Comprehensive access control patterns

// User data access pattern - Users can only access their own data
match /users/{userId}/{collection}/{document} {
  allow read, write: if request.auth != null && 
    request.auth.uid == userId;
  // Ensures authenticated users can only access their own documents
  // Prevents unauthorized access to other users' financial data
}

// Public data with creator restrictions - Read access for all, write only for creators
match /publicCategories/{document} {
  allow read: if request.auth != null; // All authenticated users can read
  allow write: if request.auth != null && (
    resource == null ||  // Allow new documents (resource is null)
    resource.data.createdBy == request.auth.uid // Only creator can modify
  );
}

// Public people collection - Similar pattern for shared people data
match /publicPeople/{document} {
  allow read: if request.auth != null;
  allow create: if request.auth != null; // Anyone can share people
  allow update, delete: if request.auth != null && 
    resource.data.userId == request.auth.uid;
}

PWA Architecture

Service Worker Implementation

Located in public/sw.js with version management:

// Service worker version tracking - Ensures cache invalidation on updates
const VERSION = '2.5.5'; // Must match package.json version
const CACHE_NAME = `budget-buddy-cache-v${VERSION}`;

// Cache strategies for different resource types
const CACHE_STRATEGIES = {
  static: 'cache-first',    // HTML, CSS, JS files
  api: 'network-first',     // Firebase API calls
  images: 'cache-first'     // Receipt images, icons
};

// PWA manager integration - Handles version updates and cache management
class PWAManager {
  currentVersion = '2.5.5'; // Synchronized with package.json
  
  async checkForVersionUpdate() {
    // Compare local version with server version
    // Triggers update notification if versions differ
    const serverVersion = await this.fetchServerVersion();
    return serverVersion !== this.currentVersion;
  }
  
  async forceAppUpdate() {
    // Clear all caches and reload app
    // Called when critical updates are available
    await this.clearAllCaches();
    window.location.reload();
  }
}

Update Notification System

Force update mechanism with user notifications:

// UpdateNotification component pattern - Handles app version management
const UpdateNotification = () => {
  const [updateType, setUpdateType] = useState<'none' | 'available' | 'force'>('none');
  
  useEffect(() => {
    const checkVersion = async () => {
      // Check for app updates periodically
      const hasUpdate = await pwaManager.checkForVersionUpdate();
      if (hasUpdate) {
        // Determine update urgency based on version difference
        const isForceUpdate = hasUpdate.isBreaking || hasUpdate.isCritical;
        setUpdateType(isForceUpdate ? 'force' : 'available');
      }
    };
    
    // Check immediately and then hourly
    checkVersion();
    const interval = setInterval(checkVersion, 60 * 60 * 1000); // Check every hour
    
    return () => clearInterval(interval);
  }, []);
  
  // Force updates block the UI until user updates
  // Available updates show notification but allow continued use
  return (
    <>
      {updateType === 'force' && <ForceUpdateModal />}
      {updateType === 'available' && <UpdateAvailableToast />}
    </>
  );
};

Performance Considerations

React Optimization

  • React.memo: Used for expensive components
  • useCallback: Memoize event handlers
  • useMemo: Expensive calculations
  • Lazy Loading: Code splitting for large components

Firebase Optimization

  • Real-time Subscriptions: Used sparingly to avoid excessive reads
  • Query Optimization: Proper indexing and filtering
  • Offline Persistence: Enabled for better performance
  • Batch Operations: Multiple writes in single transaction
