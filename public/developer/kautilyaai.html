Message Interface

// ChatMessage interface - Structured AI conversation data interface ChatMessage { id: string; // Unique message identifier role: 'user' | 'assistant'; // Message sender type content: string; // Message text content timestamp: Date; // When message was sent sessionId: string; // Session this message belongs to actionItems?: any[]; // AI-executed actions (CRUD operations) isError?: boolean; // Flag for error messages metadata?: { // AI enhancement metadata dataCount?: number; // Number of data points analyzed filters?: string[]; // Applied filters or criteria totalAmount?: number; // Financial totals calculated dateRange?: string; // Date range analyzed }; intentType?: 'DATA_RETRIEVAL' | 'ANALYSIS' | 'CRUD_OPERATION' | 'GENERAL_CHAT'; responseFormat?: 'RAW_DATA' | 'ANALYSIS' | 'CONFIRMATION' | 'CONVERSATIONAL'; } // ChatSession interface - Organized conversation sessions interface ChatSession { id: string; // Unique session identifier title: string; // Session title (auto-generated from first message) lastMessage: string; // Preview of last message timestamp: Date; // Last activity timestamp messageCount: number; // Total messages in session userId: string; // Session owner }

// Session management in AIChatPage const createNewSession = async () => { const newSessionId = `session_${Date.now()}`; const newSession: ChatSession = { id: newSessionId, title: 'New Conversation', lastMessage: '', timestamp: new Date(), messageCount: 0, userId: user.uid }; try { // Save session to Firestore await addDoc( collection(db, 'chatSessions', user.uid, 'sessions'), newSession ); setCurrentSessionId(newSessionId); setSessions(prev => [newSession, ...prev]); setMessages([]); // Clear current messages } catch (error) { console.error('Error creating session:', error); toast.error('Failed to create new session'); } }; // Load session history const loadSession = async (sessionId: string) => { try { setLoading(true); // Load messages for this session const messagesQuery = query( collection(db, 'chatMessages', sessionId, 'messages'), orderBy('timestamp', 'asc') ); const messagesSnapshot = await getDocs(messagesQuery); const sessionMessages = messagesSnapshot.docs.map(doc => ({ id: doc.id, ...doc.data(), timestamp: doc.data().timestamp.toDate() })) as ChatMessage[]; setMessages(sessionMessages); setCurrentSessionId(sessionId); } catch (error) { console.error('Error loading session:', error); toast.error('Failed to load session'); } finally { setLoading(false); } };
