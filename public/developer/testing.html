Utility Functions Testing
Unit Tests

Example Test Structure

// tests/utils.test.ts
import { getAllPeople, getExpensesByDateRange, calculateCategorySpending } from '../src/lib/types';

describe('Utility Functions', () => {
  describe('getAllPeople', () => {
    test('should combine custom and public people correctly', () => {
      const customPeople = [
        { id: '1', name: 'John', color: '#000', icon: 'user', userId: 'user1', isPublic: false, createdAt: '2025-01-01', createdBy: 'User' }
      ];
      const publicPeople = [
        { id: '2', name: 'Jane', color: '#111', icon: 'user', userId: 'user2', isPublic: true, createdAt: '2025-01-01', createdBy: 'User2' }
      ];

      const result = getAllPeople(customPeople, publicPeople);
      
      expect(result).toHaveLength(2);
      expect(result[0]).toEqual(customPeople[0]);
      expect(result[1]).toEqual(publicPeople[0]);
    });

    test('should handle duplicate IDs correctly', () => {
      const customPeople = [
        { id: '1', name: 'John', color: '#000', icon: 'user', userId: 'user1', isPublic: false, createdAt: '2025-01-01', createdBy: 'User' }
      ];
      const publicPeople = [
        { id: '1', name: 'John Public', color: '#111', icon: 'user', userId: 'user2', isPublic: true, createdAt: '2025-01-01', createdBy: 'User2' }
      ];

      const result = getAllPeople(customPeople, publicPeople);
      
      expect(result).toHaveLength(1);
      expect(result[0].name).toBe('John'); // Custom takes precedence
    });
  });

  describe('calculateCategorySpending', () => {
    test('should calculate category totals correctly', () => {
      const expenses = [
        { id: '1', amount: 50, category: 'Food', description: 'Lunch', date: '2025-08-01', createdAt: '2025-08-01T12:00:00Z' },
        { id: '2', amount: 30, category: 'Food', description: 'Snack', date: '2025-08-01', createdAt: '2025-08-01T15:00:00Z' },
        { id: '3', amount: 100, category: 'Transport', description: 'Gas', date: '2025-08-01', createdAt: '2025-08-01T18:00:00Z' }
      ];

      const result = calculateCategorySpending(expenses);
      
      expect(result).toEqual({
        'Food': 80,
        'Transport': 100
      });
    });
  });
});

Example Component Test

// tests/components/ExpenseCard.test.tsx
import { render, screen, fireEvent } from '@testing-library/react';
import { ExpenseCard } from '../../src/components/ExpenseCard';

const mockExpense = {
  id: '1',
  amount: 50.00,
  category: 'Food',
  description: 'Lunch at restaurant',
  date: '2025-08-23',
  createdAt: '2025-08-23T12:00:00Z'
};

describe('ExpenseCard', () => {
  test('renders expense information correctly', () => {
    render(
      <ExpenseCard 
        expense={mockExpense} 
        viewMode="grid" 
        onEdit={jest.fn()} 
        onDelete={jest.fn()} 
      />
    );

    expect(screen.getByText('$50.00')).toBeInTheDocument();
    expect(screen.getByText('Food')).toBeInTheDocument();
    expect(screen.getByText('Lunch at restaurant')).toBeInTheDocument();
  });

  test('calls onEdit when edit button is clicked', () => {
    const mockOnEdit = jest.fn();
    
    render(
      <ExpenseCard 
        expense={mockExpense} 
        viewMode="grid" 
        onEdit={mockOnEdit} 
        onDelete={jest.fn()} 
      />
    );

    fireEvent.click(screen.getByLabelText('Edit expense'));
    expect(mockOnEdit).toHaveBeenCalledWith(mockExpense);
  });
});

Quality Assurance Checklist

üîç Code Quality Checklist

  • ESLint rules passing with no warnings
  • Prettier formatting applied consistently
  • All components have proper TypeScript interfaces
  • Error boundaries implemented for component isolation
  • Proper error handling in async operations
  • Loading states implemented for all async operations

üé® UI/UX Quality Checklist

  • Responsive design works on all screen sizes
  • Touch targets are appropriately sized (minimum 44px)
  • Color contrast meets WCAG accessibility standards
  • Keyboard navigation works for all interactive elements
  • Screen reader compatibility with proper ARIA labels
  • Loading skeletons prevent layout shift
  • Consistent spacing and typography throughout
  • Icons and visual elements are meaningful and consistent

üîê Security Quality Checklist

  • Firebase security rules properly configured
  • User authentication required for all data operations
  • API keys properly secured and not exposed in client code
  • File upload validation and size limits enforced
  • XSS prevention measures in place
  • HTTPS enforced for all communications
  • Sensitive data not logged or exposed
  • Proper session management and logout functionality
